/*
 * @brief Customized function which takes in a graph in CSR format
 * @param[in]  num_nodes   Number of veritces in the input graph
 * @param[in]  num_edges   Number of edges in the input graph
 * @param[in]  row_offsets CSR-formatted graph input row offsets
 * @param[in]  col_indices CSR-formatted graph input column indices
 * @param[in]  edge_values CSR-formatted graph input edge weights
 * @param[in]  num_runs    Number of runs to perform SSSP
 * @param[in]  sources     Sources to begin traverse, one for each run
 * @param[in]  mark_preds  Whether to output predecessor info
 * @param[out] distances   Return shortest distance to source per vertex
 * @param[out] preds       Return predecessors of each vertex
 * @param[in]  memspace    Location of input and desired output
 * @param[in]  exec_policy Location where app will be run
 * \return     double      Return accumulated elapsed times for all runs
 */
 template <typename VertexT = unsigned long long, typename SizeT = unsigned long long,
 typename GValueT = unsigned int, typename SSSPValueT = GValueT>
double gunrock_sssp_CSR(const SizeT num_nodes, const SizeT num_edges,
   const SizeT *row_offsets, const VertexT *col_indices,
   const GValueT *edge_values, const int num_runs, VertexT *sources,
   const bool mark_pred, SSSPValueT **distances, VertexT **preds = NULL, 
   gunrock::util::Location memspace = gunrock::util::HOST,
   gunrock::util::Location exec_policy = gunrock::util::DEVICE) {
typedef typename gunrock::app::TestGraph<VertexT, SizeT, GValueT,
                                  gunrock::graph::HAS_EDGE_VALUES |
                                      gunrock::graph::HAS_CSR>
GraphT;
typedef typename GraphT::CsrT CsrT;

// Setup parameters
gunrock::util::Parameters parameters("sssp");
gunrock::graphio::UseParameters(parameters);
gunrock::app::sssp::UseParameters(parameters);
gunrock::app::UseParameters_test(parameters);
parameters.Parse_CommandLine(0, NULL);
parameters.Set("graph-type", "by-pass");
parameters.Set("mark-pred", mark_pred);
parameters.Set("num-runs", num_runs);
parameters.Set("mem-space", memspace);
parameters.Set("exec-policy", exec_policy);
parameters.Set("quiet", true);
std::vector<VertexT> srcs;
for (int i = 0; i < num_runs; i++) srcs.push_back(sources[i]);
parameters.Set("srcs", srcs);

bool quiet = parameters.Get<bool>("quiet");
GraphT graph;
// Assign pointers into gunrock graph format
graph.CsrT::Allocate(num_nodes, num_edges, memspace);
graph.CsrT::row_offsets.SetPointer((SizeT *)row_offsets, num_nodes + 1,
                            memspace);
graph.CsrT::column_indices.SetPointer((VertexT *)col_indices, num_edges,
                               memspace);
graph.CsrT::edge_values.SetPointer((GValueT *)edge_values, num_edges,
                            memspace);
graph.FromCsr(graph.csr(), memspace, 0, quiet, true);
gunrock::graphio::LoadGraph(parameters, graph);

double elapsed_time = gunrock_sssp(parameters, graph, distances, preds);

// Cleanup
graph.Release();
srcs.clear();

return elapsed_time;
}

/*
 * @brief Customized function which takes in a graph in COO format
 * @brief EdgePairT is a struct of two uint64_t values (x & y), 16 bit alligned!
 * @param[in]  num_nodes   Number of veritces in the input graph
 * @param[in]  num_edges   Number of edges in the input graph
 * @param[in]  edge_coord  COO-formatted graph input edge coordinates
 * @param[in]  edge_values COO-formatted graph input edge weights
 * @param[in]  source      Source to begin traverse
 * @param[in]  mark_preds  Whether to output predecessor info
 * @param[out] distances   Return shortest distance to source per vertex
 * @param[out] preds       Return predecessors of each vertex
 * \return     double      Return accumulated elapsed time
 */
/*template <typename VertexT = uint64_t, typename SizeT = uint64_t,
          typename GValueT = unsigned int, typename SSSPValueT = GValueT>*/
          
template <typename VertexT = unsigned long long, typename SizeT = unsigned long long,
          typename GValueT = unsigned int, typename SSSPValueT = GValueT>
double gunrock_sssp_COO(const SizeT num_nodes, const SizeT num_edges, 
  const void *edge_coords, const GValueT *edge_values, VertexT source,
  const bool mark_pred, SSSPValueT **distances, VertexT **preds, 
  gunrock::util::Location memspace = gunrock::util::HOST,
  gunrock::util::Location exec_policy = gunrock::util::DEVICE){

  typedef typename gunrock::app::TestGraph<VertexT, SizeT, GValueT,
                                    gunrock::graph::HAS_EDGE_VALUES |
                                        gunrock::graph::HAS_COO>
  GraphCooT;

  typedef typename gunrock::app::TestGraph<VertexT, SizeT, GValueT,
                                    gunrock::graph::HAS_EDGE_VALUES |
                                        gunrock::graph::HAS_CSR>
  GraphCsrT;
  typedef typename GraphCooT::CooT CooT;
  typedef typename GraphCsrT::CsrT CsrT;

  // Setup parameters
  gunrock::util::Parameters parameters("sssp");
  gunrock::graphio::UseParameters(parameters);
  gunrock::app::sssp::UseParameters(parameters);
  gunrock::app::UseParameters_test(parameters);
  parameters.Parse_CommandLine(0, NULL);
  parameters.Set("graph-type", "by-pass");
  parameters.Set("mark-pred", mark_pred);
  parameters.Set("num-runs", 1);
  parameters.Set("mem-space", memspace);
  parameters.Set("exec-policy", exec_policy);
  parameters.Set("quiet", true);
  std::vector<uint64_t> srcs;
  srcs.push_back(source);
  parameters.Set("srcs", srcs);

  bool quiet = parameters.Get<bool>("quiet");
  GraphCooT graphCoo;
  GraphCsrT graphCsr;

  graphCoo.CooT::Allocate(num_nodes, num_edges, memspace);
  graphCoo.CooT::edge_pairs.SetPointer((ulonglong2*) edge_coords, num_edges, memspace);
  graphCoo.CooT::edge_values.SetPointer((GValueT *) edge_values, num_edges,memspace);

  graphCsr.FromCoo(graphCoo.coo(), memspace, 0, quiet, true);
  gunrock::graphio::LoadGraph(parameters, graphCsr);

  // Run the SSSP
  double elapsed_time = gunrock_sssp(parameters, graphCsr, distances, preds);

  // Cleanup
  graphCoo.Release();
  graphCsr.Release();
  srcs.clear();

  return elapsed_time;
}

// Public interface
double gunrock_sssp_COO(
  const unsigned long long        num_nodes, 
  const unsigned long long        num_edges, 
  const void                      *edge_coords, 
  const float                     *edge_values, 
  unsigned long long              source,
  const bool                      mark_pred, 
  float                           *distances, 
  unsigned long long              *preds){

    gunrock::util::Location memspace = gunrock::util::HOST;
    gunrock::util::Location exec_policy = gunrock::util::DEVICE;
    return gunrock_sssp_COO(num_nodes, num_edges, edge_coords, edge_values, 
              source, mark_pred, &distances, &preds, memspace, exec_policy);
}

// Public interface
double gunrock_sssp_CSR(
  const unsigned long long      num_nodes, 
  const unsigned long long      num_edges, 
  const unsigned long long      *row_offsets,
  const unsigned long long      *col_indices, 
  const float                   *edge_values, 
  unsigned long long            source,
  const bool                    mark_pred, 
  float                         *distances, 
  unsigned long long            *preds){

  gunrock::util::Location memspace = gunrock::util::HOST;
  gunrock::util::Location exec_policy = gunrock::util::DEVICE;

  return gunrock_sssp_CSR(num_nodes, num_edges, row_offsets, col_indices, edge_values,
    1 /* num_runs */, &source, mark_pred, &distances, &preds, memspace, exec_policy);
}
